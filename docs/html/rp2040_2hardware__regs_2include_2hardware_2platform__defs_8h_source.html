58// PICO_CONFIG: XOSC_HZ, Crystal oscillator frequency in Hz, type=int, default=12000000, advanced=true, group=hardware_base
59// NOTE: The system and USB clocks are generated from the frequency using two PLLs.
60// If you override this define, or SYS_CLK_HZ/USB_CLK_HZ below, you will *also* need to add your own adjusted PLL set-up defines to
61// override the defaults which live in src/rp2_common/hardware_clocks/include/hardware/clocks.h
62// Please see the comments there about calculating the new PLL setting values.
63#ifndef XOSC_HZ
64#ifdef XOSC_KHZ
65#define XOSC_HZ ((XOSC_KHZ) * _u(1000))
66#elif defined(XOSC_MHZ)
67#define XOSC_HZ ((XOSC_MHZ) * _u(1000000))
68#else
69#define XOSC_HZ _u(12000000)
70#endif
71#endif
