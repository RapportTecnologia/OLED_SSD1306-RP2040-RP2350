87static inline void busy_wait_at_least_cycles(uint32_t minimum_cycles) {
88 pico_default_asm_volatile (
89#ifdef __riscv
90 // Note the range is halved on RISC-V due to signed comparison (no carry flag)
91 ".option push\n"
92 ".option norvc\n" // force 32 bit addi, so branch prediction guaranteed
93 ".p2align 2\n"
94 "1: \n"
95 "addi %0, %0, -2 \n"
96 "bgez %0, 1b\n"
97 ".option pop"
98#else
99 "1: subs %0, #3\n"
100 "bcs 1b\n"
101#endif
102 : "+r" (minimum_cycles) : : "cc", "memory"
103 );
104}

139static __force_inline uint __get_current_exception(void) {
140#ifdef __riscv
141 uint32_t meicontext;
142 pico_default_asm_volatile (
143 "csrr %0, %1\n"
144 : "=r" (meicontext) : "i" (RVCSR_MEICONTEXT_OFFSET)
145 );
146 if (meicontext & RVCSR_MEICONTEXT_NOIRQ_BITS) {
147 return 0;
148 } else {
149 return VTABLE_FIRST_IRQ + (
150 (meicontext & RVCSR_MEICONTEXT_IRQ_BITS) >> RVCSR_MEICONTEXT_IRQ_LSB
151 );
152 }
153#else
154 uint exception;
155 pico_default_asm_volatile (
156 "mrs %0, ipsr\n"
157 "uxtb %0, %0\n"
158 : "=l" (exception)
159 );
160 return exception;
161#endif
162}

169__force_inline static bool pico_processor_state_is_nonsecure(void) {
170#ifndef __riscv
171 // todo add a define to disable NS checking at all?
172 // IDAU-Exempt addresses return S=1 when tested in the Secure state,
173 // whereas executing a tt in the NonSecure state will always return S=0.
174 uint32_t tt;
175 pico_default_asm_volatile (
176 "movs %0, #0\n"
177 "tt %0, %0\n"
178 : "=r" (tt) : : "cc"
179 );
180 return !(tt & (1u << 22));
181#else
182 // NonSecure is an Arm concept, there is nothing meaningful to return
183 // here. Note it's not possible in general to detect whether you are
184 // executing in U-mode as, for example, M-mode is classically
185 // virtualisable in U-mode.
186 return false;
187#endif
188}
