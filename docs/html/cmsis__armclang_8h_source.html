282__STATIC_FORCEINLINE int32_t __SSAT(int32_t val, uint32_t sat)
283{
284 if ((sat >= 1U) && (sat <= 32U))
285 {
286 const int32_t max = (int32_t)((1U << (sat - 1U)) - 1U);
287 const int32_t min = -1 - max ;
288 if (val > max)
289 {
290 return (max);
291 }
292 else if (val < min)
293 {
294 return (min);
295 }
296 }
297 return (val);
298}

308__STATIC_FORCEINLINE uint32_t __USAT(int32_t val, uint32_t sat)
309{
310 if (sat <= 31U)
311 {
312 const uint32_t max = ((1U << sat) - 1U);
313 if (val > (int32_t)max)
314 {
315 return (max);
316 }
317 else if (val < 0)
318 {
319 return (0U);
320 }
321 }
322 return ((uint32_t)val);
323}
