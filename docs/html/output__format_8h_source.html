21template<typename T>
22struct simple_optional {
23 T value;
24 T default_value;
25 bool specified;
26
27 simple_optional() : default_value(), specified(false) {}
28
29 simple_optional(const T &value) : value(value), specified(true) {}
30
31 simple_optional<T> &operator=(const T &v) {
32 value = v;
33 specified = true;
34 return *this;
35 }
36
37 operator bool() = delete; // confusing
38 const T &get() const { return specified ? value : default_value; }
39
40 bool is_specified() const { return specified; }
41
42 static simple_optional<T> with_default(const T &default_value) {
43 simple_optional<T> rc;
44 rc.default_value = default_value;
45 return rc;
46 }
47};
