757__STATIC_INLINE void __NVIC_EnableIRQ(IRQn_Type IRQn)
758{
759 if ((int32_t)(IRQn) >= 0)
760 {
761 __COMPILER_BARRIER();
762 NVIC->ISER[0U] = (uint32_t)(1UL << (((uint32_t)IRQn) & 0x1FUL));
763 __COMPILER_BARRIER();
764 }
765}

776__STATIC_INLINE uint32_t __NVIC_GetEnableIRQ(IRQn_Type IRQn)
777{
778 if ((int32_t)(IRQn) >= 0)
779 {
780 return((uint32_t)(((NVIC->ISER[0U] & (1UL << (((uint32_t)IRQn) & 0x1FUL))) != 0UL) ? 1UL : 0UL));
781 }
782 else
783 {
784 return(0U);
785 }
786}
